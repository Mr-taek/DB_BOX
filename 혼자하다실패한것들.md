CREATE TABLE ALT_PRACC AS 
SELECT E2.ENAME,E2.JOB FROM EMP E1,EMP E2, SALGRADE S WHERE E1.MGR=E2.EMPNO AND E1.SAL<E2.SAL;
SELECT E2.ENAME,E2.JOB,E1.ENAME,E1.JOB FROM EMP E1,EMP E2,SALGRADE S  
WHERE E1.MGR=E2.EMPNO 
AND 3<((SELECT GRADE  FROM S WHERE E1.SEL BETWEEN S.LOSAL AND S.HISAL));
